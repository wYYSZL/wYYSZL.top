队列

队列在代码中的两种表示方法：
1.数组模拟

2.stl queue

我们用hd(head)变量表示队头，用tl(tail)表示队尾,q[]表示队列，则：

初始状态是hd=tl=1,将q[1]=a1
入队操作是t++,q[t]=入队成员

出队操作是h++

判断队列不空是while(h<=t){}

将123分别入队,并按照先进先出的顺序输出

const int MAXN = 101;
int q[MAXN];
int hd, tl;
int main()
{
    hd = 1; //初始值
    tl= 1; //初始值
    q[1] = 1;//第一个元素入队
    tl++; q[tl] = 2;//队尾++，第二个元素入队
    tl++; q[tl] = 3;//队尾++，第三个元素入队
while(hd <=tl) //队列非空 
    {
	int x = q[hd]; //去队首
	 cout << x << " ";//输出
	 hd++;  //出队操作
    }
     return 0; 
}


1443马的遍历
#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int qa[160005],qb[160005],vis[405][405],sl[405][405];
int mx[15]={1,2,-1,-2,-1,-2,1,2};
int my[15]={2,1,2,1,-2,-1,-2,-1};
int n,m,s,t;
void bfs(){
	int hd=1,tl=1;//这样写不会和s,t搞混 
	qa[hd]=s;
	qb[hd]=t;
	vis[s][t]=1;
	sl[s][t]=0;
	while(hd<=tl){
		for(int i=0;i<=7;i++){
			int xx=qa[hd]+mx[i];
			int yy=qb[hd]+my[i];
			if(!vis[xx][yy] && xx>=1 && xx<=n && yy>=1 && yy<=m){
				vis[xx][yy]=1;
				tl++;//队尾+1 
				qa[tl]=xx;
				qb[tl]=yy;
				sl[xx][yy]=sl[qa[hd]][qb[hd]]+1;
			}
		}
		hd++;
	}
}
int main()
{
	scanf("%d%d%d%d",&n,&m,&s,&t);
	memset(sl,0xff,sizeof sl);
                 //不能到达则输出-1,所以初始化为-1 
	bfs();
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++)
			printf("%-5d",sl[i][j]);
		printf("\n");
	}
}

int mx[15]={1,2,-1,-2,-1,-2,1,2};

queue<int> q1;
queue<int> q2;

for(int i=1;i<=n;i++){

void bfs(){
  q1.push(s);
  q2.push(t); vis[s][t] = 1; d[s][t] = 0;
  while(!q1.empty() && !q2.empty()){
    int ux = q1.front(); q1.pop();
    int uy = q2.front(); q2.pop();
    for(int i = 0; i <= 7; i++){
      int vx = ux + mx[i];
      int vy = uy + my[i];
      if(vx >= 1 && vx <= n && vy>=1 && vy <= m && !vis[vx][vy]){
        vis[vx][vy] = 1;
        d[vx][vy] = d[ux][uy] + 1;
        q1.push(vx);
        q2.push(vy);
      }
    }
  }
}

int movx[4]={0,1,-1,0};
int movy[4]={1,0,0,-1};
void bfs(int x,int y){
int hd=1,tl=1;
p[hd]=x;
q[hd]=y;
int xx,yy;
while(hd<=tl){
for(int i=0;i<=3;i++){
xx=p[hd]+movx[i];//一定不要忘了是队首 
yy=q[hd]+movy[i];
if(a[xx][yy] && xx>=1 && xx<=n && yy>=1 && yy<=m){
a[xx][yy]=0;
tl++;
p[tl]=xx;
q[tl]=yy;
}
}
	hd++;
}
}
